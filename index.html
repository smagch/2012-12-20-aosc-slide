UglifyJSをインストール
$ npm install uglify-js -g
UGLIFYJS_BIN = ./node_modules/.bin/uglifyjs
JS_FINAL = ./js/app.js
JS_TARGETS = \
    ./js/lib/jquery.js \
    ./js/hoge.js \
    ./js/foo.js \
    ./js/bar.js

JS_TARGETS_MIN = $(JS_TARGETS:.js=.min.js)
build: $(JS_FINAL)
    @echo "built all"

%.min.js: %.js
    @$(UGLIFYJS_BIN) -m < $< > $@
    @echo "minified $<"

$(JS_FINAL): $(JS_TARGETS_MIN)
    @cat $^ > $@

clean:
    @rm $(JS_FINAL)
.INTERMEDIATE: $(JS_TARGETS_MIN)
.PHONY: clean
